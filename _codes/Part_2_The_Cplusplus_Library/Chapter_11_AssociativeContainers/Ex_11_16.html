#include <iostream>
#include <map>
#include <vector>

int main(){
    std::map<int, std::vector<int>> i_ivec{{23, {2,3}},{32, {3,2}}};
    auto iter = i_ivec.begin();
    iter->second.push_back(5);
    std::cout << iter->first << " {";
    for(auto n : iter->second){
        std::cout << " " << n; 
    }
    std::cout << "}" << std::endl;
    
}
