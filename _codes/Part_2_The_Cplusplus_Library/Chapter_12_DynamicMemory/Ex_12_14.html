#include <iostream>
#include <memory>

struct Connection;
struct Destination;

Connection connect(destination *d_param);

void close_connection(Connection *c){
    disconnect(*c);
}

int main(){
    Destination d;
    Connection new_connect = connect(&d)
    std::shared_ptr<Connection> *c_ptr(&new_connect, close_connection);

    return 0;
}
