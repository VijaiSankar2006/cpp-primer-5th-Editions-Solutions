#include <iostream>
#include <list>                 

int main(){
    
    std::list<std::string> strlst;
    std::cout << "Enter the list of words : \n";
    std::string word;
    while(std::cin >> word){
        strlst.push_back(word);
    }

    for(auto iter = strlst.cbegin(); iter != strlst.cend(); ++iter){
        std::cout << *iter << " ";
    }

    return 0;
}
