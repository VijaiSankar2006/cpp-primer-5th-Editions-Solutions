#include <iostream>
#include <vector>

int main(){
    std::vector<int> ivec;
    std::cout << "size : " << ivec.size() << "  capacity : " << ivec.capacity() << std::endl;

    for(auto i = 0; i < 50 ; ++i){
        ivec.push_back(0);
        std::cout << "size : " << ivec.size() << "  capacity : " << ivec.capacity() << std::endl;

    }

    return 0;
}

//  vector's capacity gets doubled whenever we added a new element beyond its size.
